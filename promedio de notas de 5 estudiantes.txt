//Guardar en un arreglo las notas de 5 estudiantes.
//Calcular el promedio usando un bucle.

fun main() {
     
    val notas = arrayOf(8.5, 9.0, 7.5, 6.0, 10.0) // Notas de 5 estudiantes
    
    
    var suma = 0.0

    for (nota in notas) {
        suma += nota
    }

    val promedio = suma / notas.size

    println("El promedio es: $promedio")
}
